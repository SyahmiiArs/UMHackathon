```python
# app.py - YOU OWN THIS FILE

import streamlit as st
from glm_integration import GLMIntegration

# Page config
st.set_page_config(page_title="BudgetBrain", page_icon="🧠")

# Header
st.title("🧠 BudgetBrain")
st.caption("GLM-Powered Financial Decision Intelligence | Domain 2")

# Input form (6 fields)
col1, col2 = st.columns(2)
with col1:
    income = st.number_input("💰 Monthly Income (RM)", min_value=0, value=3000)
    rent = st.number_input("🏠 Rent (RM)", min_value=0, value=1000)
    food = st.number_input("🍜 Food (RM)", min_value=0, value=600)
with col2:
    transport = st.number_input("🚗 Transport (RM)", min_value=0, value=200)
    entertainment = st.number_input("🎬 Entertainment (RM)", min_value=0, value=300)
    others = st.number_input("📦 Others (RM)", min_value=0, value=200)

# Decision type
question = st.selectbox(
    "What decision do you need?",
    ["Where can I cut costs?", "How can I increase savings?", "Is my spending balanced?"]
)

# GLM Button
if st.button("🧠 Ask GLM", type="primary"):
    with st.spinner("GLM analyzing your finances..."):
        glm = GLMIntegration()
        result = glm.get_recommendation(income, rent, food, transport, entertainment, others, question)
    
    # Display results
    st.success("Decision generated!")
    st.markdown(result)
    
    # Accept/Reject buttons
    col_accept, col_reject = st.columns(2)
    with col_accept:
        if st.button("✅ Accept Recommendation"):
            st.balloons()
            st.success("Tracked! Expected savings recorded.")
    with col_reject:
        if st.button("❌ Reject"):
            st.info("Feedback noted.")

# Without GLM comparison (CRITICAL for judges)
with st.expander("🔍 Why GLM is essential (click to see)"):
    st.warning("""
    **Without GLM, this system would only show:**
    - Total expenses: RM X
    - Savings: RM Y
    
    **With GLM, you get:**
    - WHERE to cut specific categories
    - WHY each decision makes sense
    - HOW MUCH you will save
    - Different answers for different questions
    """)
```
